#### Read and clean precipitation data for 2017 and 2018

``` r
prec_2017 = 
  read_excel("./data/HealthyHarborWaterWheelTotals2018-7-28.xlsx",
             sheet = "2017 Precipitation",
             range = "A2:B14") %>% 
  janitor::clean_names() %>%
  mutate(year = "2017",
         month = month.name) %>% 
  drop_na() 

prec_2018 = 
  read_excel("./data/HealthyHarborWaterWheelTotals2018-7-28.xlsx",
             sheet = "2018 Precipitation",
             range = "A2:B14") %>% 
  janitor::clean_names() %>% 
  mutate(year = "2018",

  drop_na() 

#Join 2017 and 2018 percipitation
```
